35 inline bool CanScaleBeRotated(QuatArg inRotation, Vec3Arg inScale)
36 {
37 // inScale is a scale in local space of the shape, so the transform for the shape (ignoring translation) is: T = Mat44::sScale(inScale) * mRotation.
38 // when we pass the scale to the child it needs to be local to the child, so we want T = mRotation * Mat44::sScale(ChildScale).
39 // Solving for ChildScale: ChildScale = mRotation^-1 * Mat44::sScale(inScale) * mRotation = mRotation^T * Mat44::sScale(inScale) * mRotation
40 // If any of the off diagonal elements are non-zero, it means the scale / rotation is not compatible.
41 Mat44 r = Mat44::sRotation(inRotation);
42 Mat44 child_scale = r.Multiply3x3LeftTransposed(r.PostScaled(inScale));
43
44 // Get the columns, but zero the diagonal
45 Vec4 zero = Vec4::sZero();
46 Vec4 c0 = Vec4::sSelect(child_scale.GetColumn4(0), zero, UVec4(0xffffffff, 0, 0, 0)).Abs();
47 Vec4 c1 = Vec4::sSelect(child_scale.GetColumn4(1), zero, UVec4(0, 0xffffffff, 0, 0)).Abs();
48 Vec4 c2 = Vec4::sSelect(child_scale.GetColumn4(2), zero, UVec4(0, 0, 0xffffffff, 0)).Abs();
49
50 // Check if all elements are less than epsilon
51 Vec4 epsilon = Vec4::sReplicate(1.0e-6f);
52 return UVec4::sAnd(UVec4::sAnd(Vec4::sLess(c0, epsilon), Vec4::sLess(c1, epsilon)), Vec4::sLess(c2, epsilon)).TestAllTrue();
53 }

59 inline Vec3 RotateScale(QuatArg inRotation, Vec3Arg inScale)
60 {
61 // Get the diagonal of mRotation^T * Mat44::sScale(inScale) * mRotation (see comment at CanScaleBeRotated)
62 Mat44 r = Mat44::sRotation(inRotation);
63 return r.Multiply3x3LeftTransposed(r.PostScaled(inScale)).GetDiagonal3();
64 }
